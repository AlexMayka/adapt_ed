// ===== Домен 1: Школы и пользователи =====

Table schools {
  id uuid [primary key, default: `gen_random_uuid()`, note: "Идентификатор учебного заведения"]
  name text [not null, note: "Название школы"]
  city text [not null, note: "Город школы"]
  logo_key text [default: null, note: "Путь к логотипу в MinIO"]
  created_at timestamptz [not null, default: `now()`, note: "Дата создания записи"]
  updated_at timestamptz [not null, default: `now()`, note: "Дата обновления записи"]
  deleted_at timestamptz [note: "Дата удаления записи, soft delete"]

  Note: 'Учебное заведение. Верхний уровень организационной структуры. Ученик может быть без школы (индивидуальный).'
}

Table classes {
  id uuid [primary key, default: `gen_random_uuid()`, note: "Идентификатор класса"]
  school_id uuid [not null, note: "Школа класса"]
  number_of_class smallint [not null, note: "Порядковый номер класса (5, 6, 7, 8...)"]
  suffixes_of_class text [not null, note: "Суффикс класса (А, Б, В...)"]
  academic_year_start date [not null, default: '''
        CASE
            WHEN CURRENT_DATE >= make_date(extract(year from CURRENT_DATE)::int, 9, 1)
                THEN make_date(extract(year from CURRENT_DATE)::int, 9, 1)
            ELSE
                make_date(extract(year from CURRENT_DATE)::int - 1, 9, 1)
        END
    ''', note: "Дата начала учебного года"]
  academic_year_finish date [not null, default: '''
    CASE
      WHEN CURRENT_DATE > make_date(extract(year from CURRENT_DATE)::int - 1, 8, 31)
          THEN make_date(extract(year from CURRENT_DATE)::int, 8, 31)
      ELSE
          make_date(extract(year from CURRENT_DATE)::int, 8, 31)
    END
  ''', note: "Дата окончания учебного года"]
  created_at timestamptz [not null, default: `now()`, note: "Дата создания записи"]
  updated_at timestamptz [not null, default: `now()`, note: "Дата обновления записи"]
  deleted_at timestamptz [note: "Дата удаления записи, soft delete"]

  indexes {
    (school_id, number_of_class, suffixes_of_class) [unique, note: "Уникальность по школе-классу-суффиксу"]
  }

  Note: 'Класс внутри школы (7А, 7Б). Содержит точные даты учебного года для аналитики и автоматической смены класса.'
}

Ref: classes.school_id > schools.id [delete: cascade, update: no action]

Table users {
  id uuid [primary key, default: `gen_random_uuid()`, note: "Идентификатор пользователя"]
  role user_role [not null, default: 'student', note: "Роль пользователя"]
  class_id uuid [default: null, note: "Класс пользователя"]
  school_id uuid [default: null, note: "Школа пользователя"]
  email text [not null, note: "Email пользователя"]
  password_hash text [not null, note: "Хэш пароля пользователя"]
  last_name text [not null, note: "Фамилия пользователя"]
  first_name text [not null, note: "Имя пользователя"]
  middle_name text [default: null, note: "Отчество пользователя"]
  avatar_key text [default: null, note: "Путь к аватарке в MinIO"]
  is_active bool [not null, default: true, note: "Активен ли пользователь"]
  created_at timestamptz [not null, default: `now()`, note: "Дата создания записи"]
  updated_at timestamptz [not null, default: `now()`, note: "Дата обновления записи"]
  deleted_at timestamptz [note: "Дата удаления записи, soft delete"]

  indexes {
    email [unique, name: "uq_users_email", note: "Уникальный email"]
    (last_name, first_name, middle_name, school_id) [name: "idx_users_fullname", note: "Составной индекс для поиска пользователя школы"]
  }

  Note: 'Все пользователи платформы. Роль определяет возможности. school_id и class_id nullable — индивидуальные ученики без школы.'
}

Ref: users.class_id > classes.id [delete: cascade, update: no action]
Ref: users.school_id > schools.id [delete: cascade, update: no action]

Ref: student_profiles.user_id > users.id [delete: cascade, update: no action]

Table teacher_classes {
  id uuid [primary key, default: `gen_random_uuid()`, note: "Идентификатор связи учитель-класс"]
  teacher_id uuid [not null, note: "Идентификатор учителя"]
  class_id uuid [not null, note: "Идентификатор класса"]
  created_at timestamptz [not null, default: `now()`, note: "Дата создания записи"]

  indexes {

  }

Ref: teacher_classes.teacher_id > users.id [delete: cascade, update: no action]
Ref: teacher_classes.class_id > classes.id [delete: cascade, update: no action]

Table refresh_tokens {
  id uuid [primary key, default: `gen_random_uuid()`, note: "Идентификатор токена"]
  user_id uuid [not null, note: "Пользователь"]
  token_hash text [not null, note: "Хэш refresh токена"]
  device_info text [default: null, note: "Информация об устройстве (User-Agent, IP)"]
  expires_at timestamptz [not null, note: "Срок действия токена"]
  revoked_at timestamptz [default: null, note: "Дата отзыва (NULL = активен)"]
  created_at timestamptz [not null, default: `now()`, note: "Дата создания"]

  indexes {
    token_hash [unique, name: "uq_refresh_tokens_hash", note: "Быстрый поиск по хэшу"]
    user_id [name: "idx_refresh_tokens_user_id"]
    (user_id, revoked_at) [name: "idx_refresh_tokens_user_active", note: "Активные токены пользователя"]
    expires_at [name: "idx_refresh_tokens_expires", note: "Для очистки просроченных"]
  }

  Note: 'JWT refresh токены. Хранится хэш, не сам токен. device_info для отображения активных сессий. Отзыв через revoked_at без удаления.'
}

Ref: refresh_tokens.user_id > users.id [delete: cascade, update: no action]
